\def\@makechapterhead#1{%
  \vbox to 7.5cm{\parindent\z@
    \raggedright
    \reset@font
    \huge\bfseries
    \leavevmode
    \ifnum \c@secnumdepth >\m@ne
    \hskip10mm
    \vskip13.5mm
    \hrule \@height 0.8mm
    \vskip.5mm
    \hrule \@height 0.3mm
    \vskip4mm
    {\chapfont{\chaptername}}
    {\chapnumfont{\thechapter}}
    {\chapfont{\chapternamew}}
    \vskip7mm
    \vtop{\chapfontw{#1}}%
    \vskip3mm
    \hrule \@height 0.3mm
    \vskip.5mm
    \hrule \@height 0.8mm
    \vskip20mm
    \vfill
    \else
     #1\relax
    \fi}
  \nobreak
}
